Codewars task "Consecutive strings" solution:
function longestConsec(arr, k) {
  let longest = '';
  for (i = 0; i <= arr.length - k && k > 0; i++) {
      let str = arr.slice(i, i + k).join('');
      if (str.length > longest.length) longest = str;
  }
  return longest;
}
